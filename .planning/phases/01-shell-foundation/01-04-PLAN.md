    1. Helper function to run shell with input:
       ```typescript
       import { spawn } from 'node:child_process'
       import * as path from 'node:path'
       import * as fs from 'node:fs'
       import * as os from 'node:os'
       import { describe, it, expect, beforeEach, afterEach } from 'vitest'

       function runShell(input: string, options?: { cwd?: string; env?: Record<string, string> }): Promise<{ stdout: string; stderr: string; exitCode: number }> {
         return new Promise((resolve) => {
           const child = spawn('npx', ['tsx', 'src/cli.ts'], {
             cwd: options?.cwd ?? process.cwd(),
             env: { ...process.env, ...options?.env },
             stdio: ['pipe', 'pipe', 'pipe'],
           })
           let stdout = ''
           let stderr = ''
           child.stdout.on('data', (d) => { stdout += d.toString() })
           child.stderr.on('data', (d) => { stderr += d.toString() })
           child.stdin.write(input)
           child.stdin.end()
           child.on('close', (code) => {
             resolve({ stdout, stderr, exitCode: code ?? 1 })
           })
         })
       }
       ```
