       ```typescript
       import { spawn } from 'node:child_process'

       export function executeCommand(command: string, cwd?: string): Promise<number> {
         return new Promise((resolve) => {
           const child = spawn('bash', ['-c', command], {
             stdio: 'inherit',
             cwd: cwd ?? process.cwd(),
             env: process.env,
           })

           child.on('close', (code) => {
             resolve(code ?? 1)
           })

           child.on('error', (err) => {
             process.stderr.write(`Failed to execute: ${err.message}\n`)
             resolve(127)
           })
         })
       }
       ```

    2. Export async function `runShell(): Promise<void>` that:

       a. Loads history from HISTORY_PATH using loadHistory()
       b. Creates readline interface:
          ```typescript
          const rl = readline.createInterface({
            input: process.stdin,
            output: process.stdout,
            history: [...loadHistory()] as string[],
            historySize: 10_000,
            terminal: true,
          })
          ```

       c. Initializes state:
          ```typescript
          let state: ShellState = {
            cdState: { previousDir: undefined },
            running: true,
          }
          let lastHistoryLine: string | undefined
          ```

       d. Sets up SIGINT handler on rl (D-16):
          ```typescript
          rl.on('SIGINT', () => {
            // Clear current line, show fresh prompt (D-16)
            // readline handles this when using question() — SIGINT makes question() return ''
            // We just need to make sure we don't exit
          })
          ```

       e. Sets up 'close' event for Ctrl+D (D-18):
          ```typescript
          rl.on('close', () => {
            state = { ...state, running: false }
          })
          ```

       f. Main REPL while loop (D-25 — wrapped in try/catch):
          ```
          while (state.running) {
            try {
              const prompt = buildPrompt(process.cwd(), os.homedir())
              const line = await rl.question(prompt)
              const action = classifyInput(line)

              switch (action.type) {
                case 'empty':
                  break

                case 'builtin':
                  switch (action.name) {
                    case 'cd': {
                      const result = executeCd(action.args, state.cdState)
                      state = { ...state, cdState: result.newState }
                      if (result.output) process.stdout.write(result.output + '\n')
                      if (result.error) process.stderr.write(result.error + '\n')
                      break
                    }
                    case 'export': {
                      const err = executeExport(action.args)
                      if (err) process.stderr.write(err + '\n')
                      break
                    }
                    case 'clear':
                      process.stdout.write('\x1Bc')
                      break
                    case 'exit':
                    case 'quit':
                      state = { ...state, running: false }
                      break
                  }
                  break

                case 'passthrough': {
                  const exitCode = await executeCommand(action.command)
                  if (exitCode !== 0) {
                    process.stderr.write(`[exit: ${exitCode}]\n`)  // D-26
                  }
                  break
                }

                case 'ai_placeholder':
                  process.stdout.write('AI commands will be available in a future update.\n')
                  break
              }

              // History management (D-23, D-24)
              if (shouldSaveToHistory(line, lastHistoryLine)) {
                lastHistoryLine = line.trim()
              }

            } catch (err) {
              // Pitfall 3: readline question() rejects on close (Ctrl+D)
              if ((err as NodeJS.ErrnoException)?.code === 'ERR_USE_AFTER_CLOSE') break
              // D-25: Never crash from unexpected errors
              process.stderr.write(`Error: ${(err as Error).message}\n`)
            }
          }
          ```

       g. Cleanup after loop:
          ```typescript
          // Save history (D-22)
          const historyToSave = (rl as any).history ?? []
          saveHistory([...historyToSave].reverse())
          rl.close()
          ```

       h. Register process-level cleanup for TTY state (Pitfall 6):
          ```typescript
          const cleanup = () => {
            try { rl.close() } catch { /* already closed */ }
          }
          process.on('exit', cleanup)
          process.on('uncaughtException', (err) => {
            process.stderr.write(`Fatal: ${err.message}\n`)
            cleanup()
            process.exit(1)
          })
          process.on('unhandledRejection', (err) => {
            process.stderr.write(`Fatal: ${(err as Error)?.message ?? String(err)}\n`)
            cleanup()
            process.exit(1)
          })
          ```

    3. Create src/cli.ts — the entry point:

       ```typescript
       import { runShell } from './shell.js'

       runShell().catch((err) => {
         process.stderr.write(`ClaudeShell fatal error: ${(err as Error).message}\n`)
         process.exit(1)
       })
       ```
