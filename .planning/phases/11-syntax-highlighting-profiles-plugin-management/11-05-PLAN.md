    2. **Before REPL loop** (after `setupAutoSuggestions`):
       a. Start command cache for highlighting (fire-and-forget per D-04):
          `refreshCommandCache().catch(() => {})`
       b. Add registry aliases to known commands:
          `addKnownCommands(pluginRegistry.getAll().keys())`
       c. Set up highlighting keypress handler -- MUST register BEFORE suggestions per Pattern 4 in research. Since suggestions is already set up via `setupAutoSuggestions`, refactor so highlighting registers first:
          - Move `const suggestionsCleanup = setupAutoSuggestions(rl, config)` to AFTER highlighting setup
          - Add highlighting setup:
            ```
            let highlightingCleanup: (() => void) | undefined
            if (config.highlighting?.enabled !== false) {
              const handler = (_str: string | undefined, _key: object | undefined) => {
                renderHighlighted(rl, (cmd) => isKnownCommand(cmd) || pluginRegistry.resolve(cmd) !== undefined)
              }
              process.stdin.on('keypress', handler)
              highlightingCleanup = () => { process.stdin.removeListener('keypress', handler) }
            }
            ```
          - Then `const suggestionsCleanup = setupAutoSuggestions(rl, config)` AFTER
       d. First-run profile detection per D-14 (before the while loop):
          ```
          if (!safeMode) {
            const hasPluginConfig = (() => {
              try { fs.accessSync(CONFIG_PATH); return !!(loadConfig().plugins?.enabled?.length) } catch { return false }
            })()
            if (!hasPluginConfig) {
              // Show profile selector -- import executePlugin and call with 'profile' args
              await executePlugin('profile', { pluginRegistry, rl, onHotReload: (r) => {
                pluginRegistry = r.registry; hookBus = r.hookBus; enabledPlugins = r.enabled
              }})
            }
          }
          ```

    3. **Builtin switch case** (in the existing switch block that handles 'theme', 'model', etc.): Add case for `'plugin'`:
       ```
       case 'plugin': {
         await executePlugin(action.args, {
           pluginRegistry,
           rl,
           onHotReload: (r) => {
             pluginRegistry = r.registry
             hookBus = r.hookBus
             enabledPlugins = r.enabled
             // Refresh command cache with new aliases
             addKnownCommands(pluginRegistry.getAll().keys())
           },
         })
         break
       }
       ```

    4. **Cleanup** (in the cleanup section before rl.close): Call `highlightingCleanup?.()` alongside `suggestionsCleanup()`.
